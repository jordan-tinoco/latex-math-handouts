\newcommand{\polinom}[3][x]
{
    #3_{#2} #1^{#2} + #3_{#2 - 1} #1^{#2 - 1} + \cdots + #3_{1} #1 + #3_{0}
}
\newcommand{\asum}[3][x]%arithmetic sum
{
    \ifstrempty{#2}
    {
        #1_{1} +
        #1_{2}
    }
    {
        #1_{#2} +
        #1_{#2 + 1}
    }
    + \cdots + #1_{#3}
}

\newcommand{\gprod}[3][x]%product
{
    \ifstrempty{#2}
    {
        #1_{1} \cdot
        #1_{2}
    }
    {
        #1_{#2} \cdot
    #1_{#2 + 1}
    }
    \cdots #1_{#3}
}
