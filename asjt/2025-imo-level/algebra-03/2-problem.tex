\section{Problemas}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $2f(x) - 5f(y) = 8$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x) + xf(1 - x) = x$, con $x \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x - y) = f(x) + f(y) - 2xy$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x + y) + f(x)f(y) = x^2 y^2 + 2xy$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x - y) = f(x)f(y)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(f(x + y)) = x + f(y)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(f(x) + y) = 2x + f(f(y) - x)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $(y + 1)f(x)  f(x f(y) + f(x + y)) = y$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f\left(f(x)^2 + f(y)\right) = xf(x) + y$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \positiveSet{\R} \to \R$ tales que $2f(x) + f\left(\dfrac{1}{x}\right) = 3x - \dfrac{3}{x}$ para todo $x > 0$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x^2 + yf(x)) = xf(x + y)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(x) + f(f(x) - y^3) = f(x^2 + y)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f : \R \to \R$ tales que $f(f(x)) + 2f(y) = 2f(x) + 4y$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar $f: \R \to \R$ tales que $f(x^2 + f(xy)) = xf(x + y)$, con $x, y \in \R$.
\end{exercise}

\begin{exercise}
    Hallar todas las funciones $f : \Z \to \Z$ con $f(0) = 1$ que satisfacen
    \[
        f(f(n)) = f(f(n + 2) + 2) = n,
    \]
    para todo entero $n$.
\end{exercise}

\begin{exercise}
    Encontrar todas las funciones $f : \Z \to \Z$ tales que
    \[
        f(x + f(y)) = f(x) + y,\ \text{para todo}\ x,y \in \Z
    \]
\end{exercise}

\begin{exercise}
    Sea $f: (0, \infty) \to \R$ una función tal que
    \begin{enumerate}
        \item[i)] $f$ es estrictamente decreciente,
        \item[ii)] $f(x) > - \frac{1}{x}$ para todo $x > 0$ y
        \item[iii)] $f(x)f\left(f(x) + \frac{1}{x}\right) = 1$ para todo $x > 0$.
    \end{enumerate}
    Hallar $f(1)$.
\end{exercise}

\begin{exercise}
    Hallar todas las funciones $f,g : \R \to \R$ tales que $g$ es inyectiva y
    \[
        f(g(x) + y) = g(x + f(y)),\ \text{para todo}\ x,y \in \R.
    \]
\end{exercise}

\begin{problem-wos}%[India, 2010]
    Encontrar todas las funciones $f: \R \to \R$ que satisfacen
    \[
        f(x + y) + xy = f(x)f(y),\ \text{para todo}\ x,y \in \R.
    \]
\end{problem-wos}

\begin{problem-wos}%[IMO, 2002]
    Hallar todas las funciones $f: \R \to \R$ tales que, para cualesquiera $x,y,u,v$ reales, se cumple
    \[
        \left[f(x) + f(y)\right]\left[f(u) + f(v)\right] = f(xu - yv).
    \]
\end{problem-wos}

\begin{problem-wos}%[Korea, 2000]
    Hallar todas las funciones $f : \R \to \R$ que satisfacen
    \[
        f(x^2 - y^2) = (x - y)\left(f(x) + f(y)\right),\ \text{para todo}\ x,y \in \R.
    \]
\end{problem-wos}

\begin{problem-wos}%[Lista corta IMO, 1988]
    Sea $f: \N \to \N$ una función que cumple
    \[
        f(f(m) + f(n)) = m + n,\ \text{para todos}\ m,n.
    \]
    Hallar los posibles valores de $f(1988)$.
\end{problem-wos}

\begin{problem-wos}%[Lista corta IMO, 2002]
    Encontrar todas las funciones $f : \R \to \R$ tales que
    \[
        f(f(x) + y) = 2x + f(f(y) - x),\ \text{para todos}\ x,y \in \R.
    \]
\end{problem-wos}

\begin{problem-wos}%[Ibero, 1993]
    Encontrar todas las funciones estrictamente crecientes $f : \N \to \N$ que satisfacen
    \[
        f(n f(m)) = m^2 f(mn),\ \text{para todos}\ m,n \in \R.
    \]
\end{problem-wos}

\begin{problem-wos}%[Italia, 1999]
    Encontrar todas las funciones estrictamente monótonas $f : \R \to \R$ tal que
    \[
        f(x + f(y)) = f(x) + y,\ \text{para}\ x,y \in \R.
    \]
\end{problem-wos}

\begin{problem-wos}
    Hallar todas las funciones $f : \R \to \R$ tales que
    \[
        f(x f(x) + f(y)) = f(x)^2 + y.
    \]
\end{problem-wos}
